202     public static function getAll($className,$condition='',$loadManyToOne=true){
203         $objects=array();
204         $membersManyToOne=Reflexion::getMembersWithAnnotation($className, "ManyToOne");
205         $tableName=OrmUtils::getTableName($className);
206         if($condition!='')
207             $condition=" WHERE ".$condition;
208         $query=DAO::$db->query("SELECT * FROM ".$tableName.$condition);
209         Logger::log("getAll","SELECT * FROM ".$tableName.$condition);
210         foreach ($query as $row){
211             //Pour chaque enregistrement : instanciation d'un objet
212             $o=new $className();
213             $objects[]=$o;
214             foreach ($row as $k=>$v){
215                 //Modificateur et test de son existance
216                 if(!is_numeric($k)){
217                     $accesseur="set".ucfirst($k);
218                     if(method_exists($o,$accesseur)){
219                         $o->$accesseur($v);
220                     }
221                     if($loadManyToOne===true && OrmUtils::isMemberInManyToOne($className,$membersManyToOne, $k)) {
222                         DAO::getOneManyToOne($o, array($k=>$v), $membersManyToOne);
223                     }
224                 }
225             }
226             DAO::addInstanceInObjects($o);
227         }
228         return $objects;

326     public static function insertOrUpdateManyToMany($instance,$member){
327         $parser=new ManyToManyParser($instance, $member);

329             $myField=$parser->getMyFkField();
330             $field=$parser->getFkField();
331             $sql="INSERT INTO `".$parser->getJoinTable()."`(`".$myField."`,`".$field."`) VALUES (:".$myField.",:".$field.");";
332             $memberAccessor="get".ucfirst($member);
333             $memberValues=$instance->$memberAccessor();
334             $myKey=$parser->getMyPk();
335             $myAccessorId="get".ucfirst($myKey);
336             $accessorId="get".ucfirst($parser->getPk());
337             $id=$instance->$myAccessorId();
338             if(!is_null($memberValues)){
339                 DAO::$db->execute("DELETE FROM `".$parser->getJoinTable()."` WHERE `".$myField."`='".$id."'");
340                 $statement=DAO::$db->prepareStatement($sql);
341                 foreach ($memberValues as $k=>$targetInstance){
342                     $foreignId=$targetInstance->$accessorId();
343                     $foreignInstances=DAO::getAll($parser->getTargetEntity(), "`".$parser->getPk()."`"."='".$foreignId."'");
344                     if(!OrmUtils::exists($targetInstance, $parser->getPk(), $foreignInstances)){
345                         DAO::insert($targetInstance,false);
346                         $foreignId=$targetInstance->$accessorId();
347                         Logger::log("InsertMany", "Insertion d'une instance de ".get_class($instance));
348                     }
349                     DAO::$db->bindValueFromStatement($statement,$myField,$id);
350                     DAO::$db->bindValueFromStatement($statement,$field,$foreignId);
351                     $result=$statement->execute();
352                     Logger::log("InsertMany", "Insertion des valeurs dans la table association '".$parser->getJoinTable()."'");
353                 }
